# Локальный интерактивный режим

./build/course_work

# Локальный запуск SQL-файла

./build/course_work scripts/demo.sql
./build/course_work scripts/stress.sql

# Запуск сервера

./build/course_server 8080

# Проверка сервера

curl http://127.0.0.1:8080/health

# Получение токена

curl -X POST http://127.0.0.1:8080/login \
  -H 'Content-Type: application/json' \
  --data-binary '{"user":"admin","password":"admin"}'

# Отправка SQL через клиент

./build/course_client 127.0.0.1 8080 scripts/demo.sql admin admin

# Отправка SQL через curl (консоль)

TOKEN="токен"

curl -X POST http://127.0.0.1:8080/query \
  -H "Authorization: Bearer $TOKEN" \
  --data-binary 'USE demo; SELECT * FROM users;'

# Просмотр telemetry

curl http://127.0.0.1:8080/metrics

# Просмотр heartbeat

curl http://127.0.0.1:8080/heartbeat

# Просмотр логов

./build/course_logs
